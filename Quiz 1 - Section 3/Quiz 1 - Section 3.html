# Write your own code
list_csvfiles = ["data_quiz/wdi_ages1980.csv", "data_quiz/wdi_ages2000.csv", "data_quiz/wdi_ages2020.csv"]
plot_titles = ["World Countries in 1980", "World Countries in 2000", "World Countries in 2020"]

index = 0
for csvfile in list_csvfiles:
    dataset = pd.read_csv(list_csvfiles[index])
    percent_ages0to14 = dataset["percent_ages0to14"]
    percent_ages15to64 = dataset["percent_ages15to64"]

    percent_ages0to64 = percent_ages0to14 + percent_ages15to64

    plt.hist(x=percent_ages0to64)
    plt.title(plot_titles[index])
    plt.ylabel("Percent Ages 0 to 64")
    plt.xlabel("# Countries")
    plt.show()

    index = index + 1
