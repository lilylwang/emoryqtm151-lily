(e) Show a scatter plot that shows "gdp_per_capita_usd" in the x-axis and "lifexpectancy" in the y-axis.

countries = wdi["country_name"]
gdp = wdi["gdp_per_capita_usd"]
life = wdi["life_expectancy"]

for n, country in enumerate(countries):   
    plt.scatter(gdp[n], life[n], s=1000, linewidth=0.1, marker=r"${}$".format(country), edgecolor="black", alpha = 0.5)

plt.title("Life Expectancy vs GDP of countries")
plt.xlabel("GDP (per capita usd)")
